const handleLanguageChange = useCallback(async (newLocale) => {

    if (onLanguageChange) {
      onLanguageChange(newLocale)
    }
  } catch (error) {
    console.error('Failed to change language:', error)
  } finally {
    setIsChanging(false)
  }
}, [locale, isChanging, setLocale, onLanguageChange])
